CREATE TABLE `needforcarsdb`.`user` (
  `ID_USER` INT UNSIGNED NOT NULL AUTO_INCREMENT,
  `USERNAME` VARCHAR(30) NOT NULL,
  `PASSWORD` VARCHAR(30) NOT NULL,
  `PHONE_NUMBER` CHAR(13) NOT NULL,
  PRIMARY KEY (`ID_USER`),
  UNIQUE INDEX `ID_USER_UNIQUE` (`ID_USER` ASC) VISIBLE,
  UNIQUE INDEX `USERNAME_UNIQUE` (`USERNAME` ASC) VISIBLE);

CREATE TABLE `needforcarsdb`.`city` (
  `ID_CITY` INT UNSIGNED NOT NULL AUTO_INCREMENT,
  `CITY_NAME` VARCHAR(30) NOT NULL,
  PRIMARY KEY (`ID_CITY`),
  UNIQUE INDEX `ID_CITY_UNIQUE` (`ID_CITY` ASC) VISIBLE);

CREATE TABLE `needforcarsdb`.`car_type` (
  `ID_CAR_TYPE` INT UNSIGNED NOT NULL AUTO_INCREMENT,
  `CAR_TYPE` VARCHAR(20) NOT NULL,
  `CAR_COUPE` VARCHAR(30) NOT NULL,
  PRIMARY KEY (`ID_CAR_TYPE`),
  UNIQUE INDEX `ID_CAR_TYPE_UNIQUE` (`ID_CAR_TYPE` ASC) VISIBLE);

CREATE TABLE `needforcarsdb`.`car_description` (
  `ID_CAR_DESCRIPTION` INT UNSIGNED NOT NULL AUTO_INCREMENT,
  `ID_CAR_TYPE` INT UNSIGNED NOT NULL,
  `BRAND` VARCHAR(30) NOT NULL,
  `MODEL` VARCHAR(30) NOT NULL,
  `DRIVECHAIN` VARCHAR(20) NOT NULL,
  `CAR_COLOR` VARCHAR(30) NOT NULL,
  `POWER` INT NULL,
  `TRANSMISSION` CHAR(1) NULL,
  `ENGINE` VARCHAR(30) NULL,
  `DOOR_NUMBER` INT NULL,
  `WEIGHT` INT NULL,
  `FUEL` CHAR(1) NULL,
  `DISPLACEMENT` INT NULL,
  `DESCRIPTION` VARCHAR(1000) NULL,
  PRIMARY KEY (`ID_CAR_DESCRIPTION`),
  UNIQUE INDEX `ID_CAR_DESCRIPTION_UNIQUE` (`ID_CAR_DESCRIPTION` ASC) VISIBLE,
  INDEX `ID_CAR_TYPE_FK_idx` (`ID_CAR_TYPE` ASC) VISIBLE,
  CONSTRAINT `ID_CAR_TYPE_FK`
    FOREIGN KEY (`ID_CAR_TYPE`)
    REFERENCES `needforcarsdb`.`car_type` (`ID_CAR_TYPE`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);

CREATE TABLE `needforcarsdb`.`announcement` (
  `ID_ANNOUNCEMENT` INT UNSIGNED NOT NULL AUTO_INCREMENT,
  `ID_USER` INT UNSIGNED NOT NULL,
  `TITLE` VARCHAR(50) NOT NULL,
  `ANNOUNCEMENT_TYPE` CHAR(1) NOT NULL,
  `CONTACT_NAME` VARCHAR(30) NOT NULL,
  `CONTACT_NUMBER` CHAR(13) NOT NULL,
  `ID_CITY` INT UNSIGNED NOT NULL,
  `ID_CAR_DESCRIPTION` INT UNSIGNED NOT NULL,
  `PRICE` DECIMAL(15,2) UNSIGNED NOT NULL,
  PRIMARY KEY (`ID_ANNOUNCEMENT`),
  UNIQUE INDEX `ID_ANNOUNCEMENT_UNIQUE` (`ID_ANNOUNCEMENT` ASC) VISIBLE,
  INDEX `ID_USER_FK_idx` (`ID_USER` ASC) VISIBLE,
  INDEX `ID_CITY_FK_idx` (`ID_CITY` ASC) VISIBLE,
  INDEX `ID_CAR_DESCRIPITON_FK_idx` (`ID_CAR_DESCRIPTION` ASC) VISIBLE,
  CONSTRAINT `ID_USER_FK`
    FOREIGN KEY (`ID_USER`)
    REFERENCES `needforcarsdb`.`user` (`ID_USER`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `ID_CITY_FK`
    FOREIGN KEY (`ID_CITY`)
    REFERENCES `needforcarsdb`.`city` (`ID_CITY`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `ID_CAR_DESCRIPITON_FK`
    FOREIGN KEY (`ID_CAR_DESCRIPTION`)
    REFERENCES `needforcarsdb`.`car_description` (`ID_CAR_DESCRIPTION`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);
